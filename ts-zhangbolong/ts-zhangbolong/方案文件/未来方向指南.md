CREATE TABLE seats (
    id BIGSERIAL PRIMARY KEY,
    theater_id BIGINT REFERENCES theaters(id),
    performance_id BIGINT REFERENCES performances(id),
    section VARCHAR(50) NOT NULL,  -- A区、B区等
    row_number INT NOT NULL,        -- 排号
    seat_number INT NOT NULL,       -- 座位号
    seat_code VARCHAR(20) NOT NULL, -- A1、B2等
    status VARCHAR(20) NOT NULL,    -- available/occupied/locked
    price DECIMAL(10,2),
    locked_by VARCHAR(100),         -- 锁定用户ID
    locked_until TIMESTAMP,         -- 锁定到期时间
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(theater_id, performance_id, seat_code)
);

-- 创建索引
CREATE INDEX idx_seats_performance ON seats(performance_id);
CREATE INDEX idx_seats_status ON seats(status);

// SeatService.java
package com.example.theater_platform.service;

import com.example.theater_platform.entity.Seat;
import com.example.theater_platform.repository.SeatRepository;
import org.springframework.stereotype.Service;
import java.util.List;
import java.util.Map;
import java.util.stream.Collectors;

@Service
public class SeatService {
    
    private final SeatRepository seatRepository;
    
    public SeatService(SeatRepository seatRepository) {
        this.seatRepository = seatRepository;
    }
    
    /**
     * 获取演出座位状态
     */
    public Map<String, Object> getSeatStatus(Long performanceId, String section) {
        List<Seat> seats;
        if (section != null && !section.isEmpty()) {
            seats = seatRepository.findByPerformanceIdAndSection(performanceId, section);
        } else {
            seats = seatRepository.findByPerformanceId(performanceId);
        }
        
        // 按状态分组
        Map<String, Long> statusCount = seats.stream()
            .collect(Collectors.groupingBy(
                s -> s.getStatus().name(),
                Collectors.counting()
            ));
        
        // 返回座位列表和统计
        return Map.of(
            "seats", seats,
            "statistics", Map.of(
                "total", seats.size(),
                "available", statusCount.getOrDefault("AVAILABLE", 0L),
                "occupied", statusCount.getOrDefault("OCCUPIED", 0L),
                "locked", statusCount.getOrDefault("LOCKED", 0L)
            )
        );
    }
}

**步骤1：在Service中添加锁定方法**
```java
// 在SeatService.java中添加
public Map<String, Object> lockSeats(Long performanceId, List<String> seatCodes, String userId) {
    List<Seat> seats = seatRepository.findByPerformanceId(performanceId)
        .stream()
        .filter(s -> seatCodes.contains(s.getSeatCode()))
        .filter(s -> s.getStatus() == SeatStatus.AVAILABLE)
        .collect(Collectors.toList());
    
    if (seats.isEmpty()) {
        throw new RuntimeException("没有可用的座位");
    }
    
    // 锁定座位（5分钟）
    LocalDateTime lockUntil = LocalDateTime.now().plusMinutes(5);
    seats.forEach(seat -> {
        seat.setStatus(SeatStatus.LOCKED);
        seat.setLockedBy(userId);
        seat.setLockedUntil(lockUntil);
    });
    
    seatRepository.saveAll(seats);
    
    // 计算总价
    Double totalPrice = seats.stream()
        .mapToDouble(Seat::getPrice)
        .sum();
    
    return Map.of(
        "lockId", UUID.randomUUID().toString(),
        "lockedSeats", seatCodes,
        "totalPrice", totalPrice,
        "expiresAt", lockUntil
    );
}
```

**步骤1：创建同步脚本（Python示例）**
```python
# sync_to_neo4j.py
from neo4j import GraphDatabase
import psycopg2

# 连接PostgreSQL
pg_conn = psycopg2.connect(
    host="localhost",
    database="theater_db",
    user="postgres",
    password="your_password"
)

# 连接Neo4j
neo4j_driver = GraphDatabase.driver(
    "bolt://localhost:7687",
    auth=("neo4j", "your_password")
)

def sync_plays():
    """同步剧目数据"""
    pg_cursor = pg_conn.cursor()
    pg_cursor.execute("SELECT id, title, category FROM plays")
    
    with neo4j_driver.session() as session:
        for row in pg_cursor.fetchall():
            play_id, title, category = row
            session.run("""
                MERGE (p:Play {id: $id})
                SET p.title = $title,
                    p.category = $category
            """, id=str(play_id), title=title, category=category)

def sync_actors():
    """同步演员数据"""
    pg_cursor = pg_conn.cursor()
    pg_cursor.execute("SELECT id, name, role FROM actors")
    
    with neo4j_driver.session() as session:
        for row in pg_cursor.fetchall():
            actor_id, name, role = row
            session.run("""
                MERGE (a:Actor {id: $id})
                SET a.name = $name,
                    a.role = $role
            """, id=str(actor_id), name=name, role=role)

def sync_relationships():
    """同步关系数据"""
    pg_cursor = pg_conn.cursor()
    # 查询剧目-演员关系
    pg_cursor.execute("""
        SELECT play_id, actor_id, role 
        FROM play_actors
    """)
    
    with neo4j_driver.session() as session:
        for row in pg_cursor.fetchall():
            play_id, actor_id, role = row
            session.run("""
                MATCH (p:Play {id: $play_id})
                MATCH (a:Actor {id: $actor_id})
                MERGE (a)-[r:PERFORMS_IN]->(p)
                SET r.role = $role
            """, play_id=str(play_id), actor_id=str(actor_id), role=role)

if __name__ == "__main__":
    print("开始同步数据...")
    sync_plays()
    sync_actors()
    sync_relationships()
    print("同步完成！")
    
    pg_conn.close()
    neo4j_driver.close()
```

     * 算法1：相似剧目推荐（基于标签相似度）
     */
    public List<Map<String, Object>> recommendSimilarPlays(String playId) {
        String cypher = """
            MATCH (p1:Play {id: $playId})
            MATCH (p2:Play)
            WHERE p1 <> p2
            WITH p1, p2, 
                 gds.similarity.cosine(p1.tags, p2.tags) as similarity
            WHERE similarity > 0.5
            RETURN p2.id as id, 
                   p2.title as title,
                   p2.category as category,
                   p2.rating as rating,
                   similarity
            ORDER BY similarity DESC
            LIMIT 10
            """;
        
        return neo4jTemplate.query(cypher, Map.of("playId", playId))
            .all();
    }
    
    /**
     * 算法2：基于演员的推荐
     */
    public List<Map<String, Object>> recommendByActor(String actorId) {
        String cypher = """
            MATCH (a:Actor {id: $actorId})-[:PERFORMS_IN]->(p:Play)
            RETURN p.id as id,
                   p.title as title,
                   p.category as category,
                   p.rating as rating
            ORDER BY p.rating DESC
            LIMIT 10
            """;
        
        return neo4jTemplate.query(cypher, Map.of("actorId", actorId))
            .all();
    }
    
    /**
     * 算法3：基于关系的推荐（推荐相关实体）
     */
    public List<Map<String, Object>> recommendByRelationship(String playId) {
        String cypher = """
            MATCH (p:Play {id: $playId})-[*1..2]-(related)
            WHERE labels(related)[0] IN ['Play', 'Actor', 'Terminology']
            RETURN DISTINCT labels(related)[0] as type,
                   related.id as id,
                   related.title as title,
                   related.name as name
            LIMIT 20
            """;
        
        return neo4jTemplate.query(cypher, Map.of("playId", playId))
            .all();
    }
}

import com.example.theater_platform.config.AIConfig;

import com.fasterxml.jackson.databind.ObjectMapper;
import org.springframework.http.HttpHeaders;

import org.springframework.stereotype.Service;
import org.springframework.web.reactive.function.client.WebClient;
import reactor.core.publisher.Mono;

import java.util.*;

@Service
public class AINearbyService {
    
    private final WebClient webClient;
    private final AIConfig aiConfig;
    private final ObjectMapper objectMapper;
    
    public AINearbyService(AIConfig aiConfig) {
        this.aiConfig = aiConfig;
        this.objectMapper = new ObjectMapper();
        this.webClient = WebClient.builder()
            .baseUrl(aiConfig.getApiUrl())
            .defaultHeader(HttpHeaders.AUTHORIZATION, "Bearer " + aiConfig.getApiKey())
            .defaultHeader(HttpHeaders.CONTENT_TYPE, MediaType.APPLICATION_JSON_VALUE)
            .build();
    }
    
    /**
     * 生成周边推荐
     */
    public Map<String, Object> generateNearbyRecommendation(
            String theaterName,
            String theaterAddress,
            Map<String, String> userPreferences
    ) {
        // 1. 构建Prompt
        String prompt = buildPrompt(theaterName, theaterAddress, userPreferences);
        
        // 2. 调用AI API
        String aiResponse = callAI(prompt);
        
        // 3. 解析响应
        return parseAIResponse(aiResponse);
    }
    
    /**
     * 构建Prompt
     */
    private String buildPrompt(String theaterName, String theaterAddress, Map<String, String> prefs) {
        StringBuilder prompt = new StringBuilder();
        prompt.append("基于以下信息，为剧场周边生成推荐：\n\n");
        prompt.append("剧场：").append(theaterName).append("\n");
        prompt.append("位置：").append(theaterAddress).append("\n");
        
        if (prefs != null && !prefs.isEmpty()) {
            prompt.append("用户偏好：\n");
            prefs.forEach((key, value) -> 
                prompt.append("- ").append(key).append("：").append(value).append("\n")
            );
        }
        
        prompt.append("\n请推荐：\n");
        prompt.append("1. 美食推荐（3个，包含：名称、距离、价格范围、特色、推荐理由）\n");
        prompt.append("2. 交通推荐（地铁、公交、停车信息）\n");
        prompt.append("3. 好玩推荐（景点、购物、娱乐，2个，包含：名称、距离、特色）\n");
        prompt.append("\n要求：推荐理由要个性化，符合用户偏好。请以JSON格式返回，格式如下：\n");
        prompt.append("{\n");
        prompt.append("  \"food\": [{\"name\": \"\", \"distance\": \"\", \"price\": \"\", \"feature\": \"\", \"reason\": \"\"}],\n");
        prompt.append("  \"transport\": {\"subway\": \"\", \"bus\": \"\", \"parking\": \"\"},\n");
        prompt.append("  \"activities\": [{\"name\": \"\", \"distance\": \"\", \"feature\": \"\"}]\n");
        prompt.append("}");
        
        return prompt.toString();
    }
    
    /**
     * 调用AI API
     */
    private String callAI(String prompt) {
        Map<String, Object> requestBody = new HashMap<>();
        requestBody.put("model", "doubao-pro-4k");
        
        List<Map<String, String>> messages = new ArrayList<>();
        Map<String, String> userMessage = new HashMap<>();
        userMessage.put("role", "user");
        userMessage.put("content", prompt);
        messages.add(userMessage);
        
        requestBody.put("messages", messages);
        
        try {
            String response = webClient.post()
                .bodyValue(requestBody)
                .retrieve()
                .bodyToMono(String.class)
                .block();
            
            // 解析响应，提取content
            JsonNode jsonNode = objectMapper.readTree(response);
            return jsonNode.get("choices").get(0).get("message").get("content").asText();
        } catch (Exception e) {
            throw new RuntimeException("AI API调用失败", e);
        }
    }
    
    /**
     * 解析AI响应
     */
    private Map<String, Object> parseAIResponse(String aiResponse) {
        try {
            // AI返回的是JSON字符串，直接解析
            return objectMapper.readValue(aiResponse, Map.class);
        } catch (Exception e) {
            // 如果解析失败，返回默认结构
            Map<String, Object> result = new HashMap<>();
            result.put("food", new ArrayList<>());
            result.put("transport", new HashMap<>());
            result.put("activities", new ArrayList<>());
            return result;
        }
    }
}

## 二、差分隐私树（ACDP-Tree++）应用场景

### 2.1 什么是差分隐私树？

差分隐私树是一种隐私保护技术，用于在保护用户隐私的前提下，对数据进行统计分析。

**核心思想**：
- 在数据中添加噪声，保护个体隐私
- 保证统计结果的准确性
- 适用于需要发布统计数据但保护用户隐私的场景

### 2.2 应用场景分析

#### ❌ 不适合的场景（实时推荐）

1. **知识图谱推荐**
   - **原因**：推荐是基于公开数据（剧目、演员信息），不涉及用户隐私数据
   - **结论**：不需要差分隐私

2. **AI周边生成**
   - **原因**：推荐是基于公开POI数据（餐厅、景点），用户偏好是实时查询的
   - **结论**：不需要差分隐私（用户数据只在查询时使用，不存储）

3. **实时个性化推荐**
   - **原因**：需要实时响应（<2秒），差分隐私会降低精度和速度
   - **结论**：使用传统加密即可（HTTPS + 数据脱敏）

#### ✅ 适合的场景（统计数据发布）

1. **热门剧目统计发布**
   - **场景**：向公众发布"本月最受欢迎剧目Top 10"
   - **隐私风险**：如果直接统计，可能泄露某些用户的观看偏好
   - **解决方案**：
     ```python
     # 使用差分隐私树处理统计数据
     from acdp_tree import ACDPTree
     
     # 统计用户观看记录（已添加噪声）
     dp_stats = ACDPTree(
         epsilon=1.6,  # 隐私参数（论文推荐值）
         max_depth=8
     ).process(user_watching_records)
     
     # 发布统计结果（保护用户隐私）
     publish_top_plays(dp_stats)
     ```

2. **用户画像聚合统计**
   - **场景**：向合作伙伴发布"用户群体特征"（如：70%用户喜欢音乐剧）
   - **隐私风险**：可能泄露个人偏好
   - **解决方案**：使用差分隐私处理聚合数据

3. **跨平台数据共享**
   - **场景**：多个剧院平台合作，共享用户行为统计数据来训练推荐模型
   - **隐私风险**：直接共享会泄露用户隐私
   - **解决方案**：
     ```python
     # 各平台先用差分隐私处理自己的数据
     platform1_dp_data = ACDPTree(epsilon=1.6).process(platform1_data)
     platform2_dp_data = ACDPTree(epsilon=1.6).process(platform2_data)
     
     # 然后共享处理后的数据（已保护隐私）
     shared_data = combine(dp_data1, dp_data2)
     ```

4. **知识图谱数据发布**
   - **场景**：向公众发布知识图谱统计数据（如："最常被查询的演员Top 10"）
   - **隐私风险**：查询记录可能泄露用户兴趣
   - **解决方案**：对查询统计结果使用差分隐私

   - 对需要发布的统计数据使用差分隐私处理
   - 不影响实时推荐功能

### 2.4 总结

**差分隐私树的应用原则**：
- ✅ **用于**：统计数据发布、数据共享、隐私保护分析
- ❌ **不用于**：实时推荐、个人查询、业务功能

**当前阶段建议**：
- **优先实现**：知识图谱推荐、AI周边生成（不需要差分隐私）
- **后续考虑**：如果需要发布统计数据，再引入差分隐私

## 五、注意事项

1. **数据获取合法性**
   - 只使用公开数据源
   - 与剧院合作获取授权数据
   - 用户生成内容需审核

   - 实时推荐使用传统加密即可
   - 统计数据发布才需要差分隐私
   - 遵循数据保护法规

3. **性能考虑**
   - 差分隐私会降低精度和速度
   - 只用于非实时场景
   - 实时推荐不使用差分隐私

4. **成本控制**
   - 大模型API按token计费，需要控制调用次数
   - 使用缓存减少API调用
   - 优化Prompt减少token消耗
